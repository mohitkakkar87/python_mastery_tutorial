def compute_metrics(y_true, y_pred):
    """Return accuracy, precision, recall as a tuple."""
    correct = sum(a == p for a, p in zip(y_true, y_pred))
    accuracy = correct / len(y_true)
    tp = sum(a == 1 and p == 1 for a, p in zip(y_true, y_pred))
    fp = sum(a == 0 and p == 1 for a, p in zip(y_true, y_pred))
    fn = sum(a == 1 and p == 0 for a, p in zip(y_true, y_pred))
    precision = tp / (tp + fp) if (tp + fp) > 0 else 0
    recall    = tp / (tp + fn) if (tp + fn) > 0 else 0
    return accuracy, precision, recall    # returns a TUPLE

y_true = [1, 0, 1, 1, 0, 1]
y_pred = [1, 0, 0, 1, 0, 1]
acc, prec, rec = compute_metrics(y_true, y_pred)
print(f"Accuracy={acc:.2f}, Precision={prec:.2f}, Recall={rec:.2f}")

class NeuralNetwork:

        # instance attributes (unique per object)

        self._loss_history = []   # "private" by convention

        """Return architecture summary."""
        arch = " → ".join(str(n) for n in self.layers)
        return f"🧠 {self.name} | {arch} | lr={self.lr}"
    
    def train(self, epochs: int = 10) -> None:
        """Simulate training."""
        for e in range(epochs):
            loss = 1.0 / (e + 1)
            self._loss_history.append(loss)
        self.is_trained = True
        print(f"✅ Trained {self.name} for {epochs} epochs")
    
    def predict(self, inputs: List[float]) -> List[float]:
        if not self.is_trained:
            raise RuntimeError("Model must be trained before prediction!")
        return [round(x * 0.8 + 0.1, 3) for x in inputs]
    
    @classmethod
    def from_config(cls, config: dict) -> 'NeuralNetwork':
        """Alternative constructor from a config dict."""
        return cls(
            name=config["name"],
            layers=config["layers"],
            lr=config.get("lr", 0.001)
        )
    
    @staticmethod
    def activation(x: float, fn: str = "relu") -> float:
        """Apply activation function."""
        if fn == "relu":    return max(0.0, x)
        if fn == "sigmoid": return 1 / (1 + 2.718 ** (-x))
        return x
    
    def __repr__(self) -> str:
        return f"NeuralNetwork(name={self.name!r}, layers={self.layers})"
    
    def __len__(self) -> int:
        return len(self.layers)

class BaseModel:
    """Abstract base for all ML models."""
    
    def __init__(self, name: str, version: str = "1.0"):
        self.name = name
        self.version = version
        self.is_fitted = False
    
    def fit(self, X, y):
        """Train the model. Override in subclasses."""
        raise NotImplementedError("Subclass must implement fit()")
    
    def predict(self, X):
        """Predict. Override in subclasses."""
        raise NotImplementedError("Subclass must implement predict()")
    
    def save(self, path: str):
        import pickle
        with open(path, "wb") as f:
            pickle.dump(self, f)
        print(f"💾 Saved {self.name} to {path}")
    
    def __str__(self):
        return f"{self.name} v{self.version} ({'trained' if self.is_fitted else 'untrained'})"


# ── CHILD CLASS ────────────────────────────────────────────
class LinearRegression(BaseModel):
    """Simple linear regression: y = mx + b"""
    
    def __init__(self, name="LinearRegression", lr=0.01, epochs=100):
        super().__init__(name)          # call parent __init__
        self.lr = lr
        self.epochs = epochs
        self.m = 0.0
        self.b = 0.0
    
    def fit(self, X: list, y: list):
        """Gradient descent training."""
        n = len(X)
        for _ in range(self.epochs):
            y_pred = [self.m * x + self.b for x in X]
            dm = -2/n * sum((y[i] - y_pred[i]) * X[i] for i in range(n))
            db = -2/n * sum(y[i] - y_pred[i] for i in range(n))
            self.m -= self.lr * dm
            self.b -= self.lr * db
        self.is_fitted = True
        return self
    
    def predict(self, X: list) -> list:
        if not self.is_fitted:
            raise RuntimeError("Fit the model first!")
        return [round(self.m * x + self.b, 4) for x in X]

class ModelRegistry:
    """Encapsulates model storage with controlled access."""
    
    def __init__(self):
        self._models = {}          # protected (single underscore)
        self.__access_log = []     # private (name-mangled)
    
    @property
    def model_count(self) -> int:
        return len(self._models)
    
    @property
    def models(self) -> list:
        return list(self._models.keys())   # read-only copy
    
    def register(self, name: str, model) -> None:
        if name in self._models:
            raise ValueError(f"Model '{name}' already registered")
        self._models[name] = model
        self.__access_log.append(f"REGISTER: {name}")
        print(f"✅ Registered model: {name}")
    
    def get(self, name: str):
        self.__access_log.append(f"GET: {name}")
        return self._models.get(name)
    
    def audit_log(self) -> list:
        return self.__access_log.copy()

class DataConnector(ABC):
    """Abstract base — defines what connectors must implement."""
    
    @abstractmethod
    def connect(self) -> bool: pass
    
    @abstractmethod
    def read(self, query: str) -> list: pass
    
    @abstractmethod
    def write(self, data: list) -> int: pass
    
    @abstractmethod
    def disconnect(self) -> None: pass
    
    # Concrete method (optional) — uses abstract methods
    def execute_pipeline(self, query: str, transform_fn) -> list:
        self.connect()
        raw = self.read(query)
        result = [transform_fn(r) for r in raw]
        self.disconnect()
        return result

    # ── LENGTH & ITERATION ────────────────────────────────
    def __len__(self):  return len(self.data)
    def __iter__(self): return iter(self.data)
    def __getitem__(self, idx): return self.data[idx]

    def __add__(self, other):
        if isinstance(other, Tensor):
            return Tensor([a + b for a, b in zip(self.data, other.data)])
        return Tensor([x + other for x in self.data])
    
    def __mul__(self, scalar):
        return Tensor([x * scalar for x in self.data])
    
    def __rmul__(self, scalar):         # scalar * tensor
        return self.__mul__(scalar)
    
    def __matmul__(self, other):        # @ operator (dot product)
        return sum(a * b for a, b in zip(self.data, other.data))
    
    # ── COMPARISON ────────────────────────────────────────
    def __eq__(self, other): return self.data == other.data
    def __lt__(self, other): return sum(self.data) < sum(other.data)

        if activation == "relu":
            return Tensor([max(0, x) for x in self.data])
        return self

    def __bool__(self): return any(x != 0 for x in self.data)

def timer(func):
    """Decorator: measure execution time."""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"⏱️ {func.__name__} took {elapsed:.4f}s")
        return result
    return wrapper

def retry(max_attempts=3, delay=1.0, exceptions=(Exception,)):
    """Decorator: retry a function on failure."""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except exceptions as e:
                    print(f"⚠️ Attempt {attempt}/{max_attempts} failed: {e}")
                    if attempt < max_attempts:
                        time.sleep(delay)
            raise RuntimeError(f"All {max_attempts} attempts failed")
        return wrapper
    return decorator

    lg.setLevel(logging.DEBUG)
    
    formatter = logging.Formatter(
        "%(asctime)s | %(levelname)-8s | %(name)s | %(message)s",

    )

    ch.setLevel(logging.INFO)
    ch.setFormatter(formatter)
    lg.addHandler(ch)
    
    # File handler (optional)

        fh = logging.FileHandler(log_file, encoding="utf-8")
        fh.setLevel(logging.DEBUG)
        fh.setFormatter(formatter)
        lg.addHandler(fh)
    
    return lg
